Variables are like labeled boxes that store information. You can put different types of data in them, look at what's inside, and change the contents whenever you need to.

Functions & Logic: Making Decisions

Create reusable code blocks and add decision-making to your programs

What are Functions?

Functions are like recipes or instruction sets. You give them a name, define what they should do, and then you can use them over and over again by calling their name.

Function Structure

Basic Function Syntax 
// Function declaration
function functionName(parameters) {
    // Code to execute
    return result; // Optional
}

// Example: A greeting function
function greetUser(name) {
    return "Hello, " + name + "!";
}

// Using (calling) the function
let message = greetUser("Sarah");
console.log(message); // "Hello, Sarah!"

Practical Function Examples

üìß Simple Functions

// Function with no parameters
function sayHello() {
    return "Hello World!";
}

// Function with parameters
function addNumbers(a, b) {
    return a + b;
}

// Using the functions
sayHello(); // "Hello World!"
addNumbers(5, 3); // 8

üßÆ Math Functions

// Calculate area of rectangle
function calculateArea(width, height) {
    return width * height;
}

// Calculate discount price
function applyDiscount(price, percent) {
    let discount = price * (percent / 100);
    return price - discount;
}

// Usage examples
calculateArea(10, 5); // 50
applyDiscount(100, 20); // 80

Conditional Logic: Making Decisions

Sometimes your code needs to make decisions based on different conditions. JavaScript provides several ways to do this:

ü§î If Statements

// Basic if statement
let age = 18;

if (age >= 18) {
    console.log("You can vote!");
}

// If-else statement
if (age >= 18) {
    console.log("You're an adult");
} else {
    console.log("You're a minor");
}

// Multiple conditions
if (age < 13) {
    console.log("Child");
} else if (age < 18) {
    console.log("Teenager");
} else {
    console.log("Adult");
}

üîÑ Loops

// For loop - repeat a specific number of times
for (let i = 1; i <= 5; i++) {
    console.log("Count: " + i);
}

// While loop - repeat while condition is true
let count = 1;
while (count <= 3) {
    console.log("Loop: " + count);
    count++;
}

// Loop through an array
let colors = ["red", "green", "blue"];
for (let color of colors) {
    console.log("Color: " + color);
}

DOM Manipulation: Changing Web Pages

Learn how to dynamically modify HTML content and CSS styles

What is the DOM?

The DOM (Document Object Model) is like a tree representation of your HTML page that JavaScript can interact with. It allows you to find, change, add, or remove any element on the page.

DOM Tree Structure

document
html
head
title
body
h1
p
div

Essential DOM Methods

üîç Finding Elements

// Find by ID
let element = document.getElementById("myId");

// Find by class
let elements = document.getElementsByClassName("myClass");

// Find by CSS selector (modern way)
let element = document.querySelector("#myId");
let elements = document.querySelectorAll(".myClass");

These methods help you locate specific HTML elements so you can modify them.

‚úèÔ∏è Changing Content

// Change text content

// Change HTML content

// Change attributes

üé® Changing Styles

Events & Interactivity: Responding to Users

Make your web pages respond to clicks, keyboard input, and other user actions

What are Events?

Events are things that happen in the browser - like clicking a button, typing in a text field, or moving the mouse. JavaScript can "listen" for these events and run code when they happen.

Common Event Types

üëÜ
Click Events

When user clicks on something

‚å®Ô∏è
Keyboard Events

When user types or presses keys

üñ±Ô∏è
Mouse Events

Mouse movement, hover, etc.

üìù
Form Events

Form submission, input changes

Adding Event Listeners

üéØ addEventListener Method

Modern Event Handling 
// Find the element
let button = document.getElementById("myButton");

// Add click event listener
button.addEventListener("click", function() {
    alert("Button was clicked!");
});

// You can also use arrow functions
button.addEventListener("click", () => {
    console.log("Modern syntax!");
});

// Multiple events on same element
button.addEventListener("mouseenter", () => {
    button.style.backgroundColor = "lightblue";
});

button.addEventListener("mouseleave", () => {
    button.style.backgroundColor = "";
});

üìä Event Object

// Event object contains useful information
button.addEventListener("click", function(event) {
    console.log("Click position:", event.clientX, event.clientY);
    console.log("Element clicked:", event.target);
    
    // Prevent default behavior
    event.preventDefault();
});

// Keyboard events
input.addEventListener("keydown", function(event) {
    console.log("Key pressed:", event.key);
    
    if (event.key === "Enter") {
        console.log("Enter key pressed!");
    }
});

function animateSkillBars() {
    const skillBars = document.querySelectorAll('.skill-progress');

    // Theme switcher functionality
function initThemeSwitcher() {
    const themeBtn = document.getElementById('theme-toggle');
    const body = document.body;
    
    // Check for saved theme
    const savedTheme = localStorage.getItem('theme');
    if (savedTheme) {
        body.classList.add(savedTheme);
        updateThemeButton(savedTheme);
    }
    
    themeBtn.addEventListener('click', () => {
        if (body.classList.contains('dark-theme')) {
            body.classList.remove('dark-theme');
            localStorage.setItem('theme', 'light-theme');
            updateThemeButton('light-theme');
        } else {
            body.classList.add('dark-theme');
            localStorage.setItem('theme', 'dark-theme');
            updateThemeButton('dark-theme');
        }
    });
}

function updateThemeButton(theme) {
    const themeBtn = document.getElementById('theme-toggle');
    themeBtn.textContent = theme === 'dark-theme' ? '‚òÄÔ∏è Light' : 'üåô Dark';
}

// Initialize when page loads
document.addEventListener('DOMContentLoaded', initThemeSwitcher);

    Contact Form Validation

    Add form validation and feedback:

    Form Validation 
    // Contact form validation
function initContactForm() {
    const form = document.getElementById('contact-form');
    
    if (form) {
        form.addEventListener('submit', function(e) {
            e.preventDefault();
            
            const name = document.getElementById('name').value.trim();
            const email = document.getElementById('email').value.trim();
            const message = document.getElementById('message').value.trim();
            
            // Simple validation
            if (name === '' || email === '' || message === '') {
                showMessage('Please fill in all fields!', 'error');
                return;
            }
            
            if (!isValidEmail(email)) {
                showMessage('Please enter a valid email!', 'error');
                return;
            }
            
            // Success
            showMessage('Thank you! Message sent successfully!', 'success');
            form.reset();
        });
    }
}

function isValidEmail(email) {
    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    return emailRegex.test(email);
}

function showMessage(text, type) {
    const messageDiv = document.getElementById('form-message');
    messageDiv.textContent = text;
    messageDiv.className = `form-message ${type}`;
    messageDiv.style.display = 'block';
    
    setTimeout(() => {
        messageDiv.style.display = 'none';
    }, 5000);
}

document.addEventListener('DOMContentLoaded', initContactForm);

    Smooth Navigation

    Add smooth scrolling to navigation links:

    Smooth Scrolling 
    // Smooth scrolling navigation
function initSmoothScrolling() {
    const navLinks = document.querySelectorAll('nav a[href^="#"]');
    
    navLinks.forEach(link => {
        link.addEventListener('click', function(e) {
            e.preventDefault();
            
            const targetId = this.getAttribute('href').substring(1);
            const targetSection = document.getElementById(targetId);
            
            if (targetSection) {
                targetSection.scrollIntoView({
                    behavior: 'smooth',
                    block: 'start'
                });
                
                // Update active navigation
                updateActiveNav(this);
            }
        });
    });
}

function updateActiveNav(activeLink) {
    const navLinks = document.querySelectorAll('nav a');
    navLinks.forEach(link => link.classList.remove('active'));
    activeLink.classList.add('active');
}

document.addEventListener('DOMContentLoaded', initSmoothScrolling);
